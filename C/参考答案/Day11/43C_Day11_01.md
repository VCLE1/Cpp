### 题目描述

二分查找的变种：
(a) 查找最后一个与key相等的元素

```c
int binary_search(int arr[], int n, int key) {
	int left = 0, right = n - 1;
	while (left <= right) {
		int mid = left + (right - left >> 1);
		if (key < arr[mid]) {
			right = mid - 1;
		}
		else if(key > arr[mid]) {
			left = mid + 1;
		}
		else {
			if (mid == right || arr[mid + 1] > key) {
				return mid;
			}
			left = mid + 1;
		}
	}
	return -1;
}
```

(b) 查找第一个大于等于key值的元素

```c
int binary_search(int arr[], int n, int key) {
	int left = 0, right = n - 1;
	while (left <= right) {
		int mid = left + (right - left >> 1);
		if (arr[mid] < key) {
			left = mid + 1;
		}
		else {
			if (mid == left || arr[mid - 1] < key) {
				return mid;
			}
			right = mid - 1;
		}
	}
	return -1;
}
```
